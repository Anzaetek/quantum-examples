@assert unitary
@prove "bell_correct" equiv { creates }
@bound gate_count = 2

circuit Bell {
    qreg q[2]
    creg c[2]

    apply H on q[0]
    apply CX on q[0], q[1]

    measure q -> c
}

circuit Grover3(marked: int) {
    qreg q[3]
    let k = 2
    repeat k {
        apply H on q[0]
    }
    when bit(marked, 0) == 0 { apply X on q[0] }
}
